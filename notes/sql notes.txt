sql = sequencing query language

Commands:

CREATE DATABASE ______ -> Create a new database
DROP DATABASE ______ -> Deletes entire database

CREATE TABLE ______ -> Create a new table
DROP TABLE ______ -> Deletes entire table

SELECT ______ FROM ______ -> Selects information from database
ADD ______ ______ -> Adds information to database
INSERT ______ INTP ______ -> Inserts information into a database
DELETE ______ FROM ______ -> Deletes information from a database
UPDATE ______ SET ______ WHERE ______ -> Updates already existing information in database
